--- name: Choropleth Map of 2014 US Population by State language: plotly_js suite: choropleth-maps order: 5 sitemap: false arrangement: horizontal --- d3.csv('https://raw.githubusercontent.com/plotly/datasets/master/2014_usa_states.csv') .then((rows) => { const unpack = (rows, key) => rows.map((row) => row[key]); const data = [{ type: 'choropleth', locationmode: 'USA-states', locations: unpack(rows, 'Postal'), z: unpack(rows, 'Population'), text: unpack(rows, 'State'), autocolorscale: true }]; const layout = { title: {text: '2014 US Population by State'}, geo:{ scope: 'usa', countrycolor: 'rgb(255, 255, 255)', showland: true, landcolor: 'rgb(217, 217, 217)', showlakes: true, lakecolor: 'rgb(255, 255, 255)', subunitcolor: 'rgb(255, 255, 255)', lonaxis: {}, lataxis: {} } }; Plotly.newPlot("myDiv", data, layout, {showLink: false}); });
